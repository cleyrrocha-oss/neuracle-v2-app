function drawMiniBrain() {
  const c = document.getElementById('mini-canvas');
  const ctx = c.getContext('2d');
  const S = 52, cx = 26, cy = 26;
  miniT++;
  ctx.clearRect(0, 0, S, S);
  ctx.beginPath(); ctx.arc(cx, cy, 26, 0, Math.PI*2);
  ctx.fillStyle = '#0a0820'; ctx.fill();
  const hb = 1 + Math.exp(-Math.pow(((miniT%80)/80 - 0.1)/0.04, 2))*0.18;
  const r = 14 * hb;
  for(let i=0;i<10;i++) {
    const a = (i/10)*Math.PI*2 + miniT*0.008;
    const x = cx + Math.cos(a)*r;
    const y = cy + Math.sin(a)*r*0.85;
    ctx.beginPath(); ctx.moveTo(cx,cy); ctx.lineTo(x,y);
    ctx.strokeStyle = `rgba(200,220,255,${0.08*hb})`; ctx.lineWidth=0.5; ctx.stroke();
    ctx.beginPath(); ctx.arc(x,y,1.2,0,Math.PI*2);
    ctx.fillStyle = `rgba(255,255,255,${0.6*hb})`; ctx.fill();
  }
  const cg = ctx.createRadialGradient(cx,cy,0,cx,cy,8*hb);
  cg.addColorStop(0,`rgba(255,255,255,${0.9*hb})`);
  cg.addColorStop(0.5,`rgba(200,220,255,0.3)`);
  cg.addColorStop(1,'transparent');
  ctx.beginPath(); ctx.arc(cx,cy,8*hb,0,Math.PI*2);
  ctx.fillStyle=cg; ctx.fill();
  miniRAF = requestAnimationFrame(drawMiniBrain);
}

let recognition = null, isRecording = false, savedText = '';
function toggleRecording() {
  if(!('webkitSpeechRecognition' in window) && !('SpeechRecognition' in window)) {
    simulateVoice(); return;
  }
  isRecording ? stopRecording() : startRecordingNote();
}
function startRecordingNote() {
  const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
  recognition = new SR();
  recognition.lang = 'pt-BR'; recognition.continuous = true; recognition.interimResults = true;
  recognition.onstart = () => {
    isRecording = true;
    document.getElementById('micBtn').classList.add('recording');
    document.getElementById('voiceHint').textContent = 'Ouvindo… toque para salvar';
    document.getElementById('voiceTranscript').classList.add('visible');
  };
  recognition.onresult = (e) => {
    let interim = '', final = '';
    for(let i=e.resultIndex;i<e.results.length;i++) {
      if(e.results[i].isFinal) final += e.results[i][0].transcript;
      else interim += e.results[i][0].transcript;
    }
    savedText += final;
    document.getElementById('transcriptText').textContent = savedText + interim;
  };
  recognition.onerror = () => stopRecording();
  recognition.start();
}
function stopRecording() {
  if(recognition) recognition.stop();
  isRecording = false;
  document.getElementById('micBtn').classList.remove('recording');
  if(savedText.trim()) {
    document.getElementById('voiceHint').textContent = '✓ Nota salva';
    setTimeout(() => {
      saveNote(savedText.trim());
      savedText = '';
      document.getElementById('transcriptText').textContent = '';
      document.getElementById('voiceTranscript').classList.remove('visible');
      document.getElementById('voiceHint').textContent = 'Toque no microfone\ne fale sua nota';
    }, 1200);
  } else {
    document.getElementById('voiceHint').textContent = 'Toque no microfone\ne fale sua nota';
  }
}
function simulateVoice() {
  const demos = ['Registrar o domínio neuracle.com.br ainda hoje.','Interface neural pulsa como coração.','Onboarding conversacional — Neura aprende a rotina.'];
  const text = demos[Math.floor(Math.random()*demos.length)];
  document.getElementById('voiceTranscript').classList.add('visible');
  document.getElementById('micBtn').classList.add('recording');
  document.getElementById('voiceHint').textContent = 'Simulando voz...';
  let i=0;
  const iv = setInterval(() => {
    i++;
    document.getElementById('transcriptText').textContent = text.slice(0,i*3);
    if(i*3>=text.length) {
      clearInterval(iv);
      setTimeout(() => {
        saveNote(text);
        document.getElementById('micBtn').classList.remove('recording');
        document.getElementById('transcriptText').textContent = '';
        document.getElementById('voiceTranscript').classList.remove('visible');
        document.getElementById('voiceHint').textContent = 'Toque no microfone\ne fale sua nota';
      }, 900);
    }
  }, 40);
}

  const val = inp.value.trim();
  if(!val) return;
  const dialog = document.getElementById('ob-dialog');
  const row = document.getElementById('ob-input-row');
  if(row) row.remove();
  const ub = document.createElement('div');
  ub.className = 'ob-bubble user';
  ub.textContent = val;
  dialog.appendChild(ub);
  dialog.scrollTop = dialog.scrollHeight;
  const step = obSteps[obStep];
  const reply = step.process(val);
  obStep++;
  setTimeout(() => {
    if(obStep >= obSteps.length) {
      const fb = document.createElement('div');
      fb.className = 'ob-bubble neura';
      fb.style.whiteSpace = 'pre-line';
      fb.textContent = reply + '\n\nTudo pronto. Sou sua Neura.';
      dialog.appendChild(fb);
      dialog.scrollTop = dialog.scrollHeight;
      setTimeout(finishOnboarding, 2200);
    } else {
      const rb = document.createElement('div');
      rb.className = 'ob-bubble neura';
      rb.style.whiteSpace = 'pre-line';
      rb.textContent = reply;
      dialog.appendChild(rb);
      dialog.scrollTop = dialog.scrollHeight;
      setTimeout(showObStep, 800);
    }
  }, 600);
}
function finishOnboarding() {
  const ob = document.getElementById('screen-onboarding');
  ob.style.transition = 'opacity 0.6s ease';
  ob.style.opacity = '0';
  setTimeout(() => {
    ob.classList.remove('visible');
    ob.style.opacity = '';
    ob.style.transition = '';
    stopObCanvas();
  }, 600);
}

let obT = 0, obRAF = null;
function startObCanvas() { drawObCanvas(); }

function drawObCanvas() {
  const c = document.getElementById('ob-canvas');
  const ctx = c.getContext('2d');
  const S=200, cx=100, cy=100;
  obT++;
  ctx.clearRect(0,0,S,S);
  const hb = 1+Math.exp(-Math.pow(((obT%90)/90-0.1)/0.04,2))*0.2;
  const maxR = 60*hb;
  for(let i=0;i<4;i++) {
    const rf=0.3+i*0.2, wave=Math.sin(obT*0.01+i*1.2)*0.5+0.5;
    ctx.beginPath(); ctx.arc(cx,cy,maxR*rf,0,Math.PI*2);
    ctx.strokeStyle=`rgba(200,220,255,${0.05*wave})`; ctx.lineWidth=0.8; ctx.stroke();
  }
  for(let i=0;i<16;i++) {
    const a=(i/16)*Math.PI*2+obT*0.005;
    const r=(i%2===0?maxR*0.55:maxR*0.85);
    const x=cx+Math.cos(a)*r, y=cy+Math.sin(a)*r*0.88;
    ctx.beginPath(); ctx.moveTo(cx,cy); ctx.lineTo(x,y);
    ctx.strokeStyle=`rgba(200,220,255,${0.06*hb})`; ctx.lineWidth=0.5; ctx.stroke();
    ctx.beginPath(); ctx.arc(x,y,1.5,0,Math.PI*2);
    ctx.fillStyle=`rgba(255,255,255,${0.55*hb})`; ctx.fill();
  }
  const cg=ctx.createRadialGradient(cx,cy,0,cx,cy,16*hb);
  cg.addColorStop(0,`rgba(255,255,255,${0.95*hb})`);
  cg.addColorStop(0.5,`rgba(200,220,255,0.4)`);
  cg.addColorStop(1,'transparent');
  ctx.beginPath(); ctx.arc(cx,cy,16*hb,0,Math.PI*2); ctx.fillStyle=cg; ctx.fill();
  obRAF = requestAnimationFrame(drawObCanvas);
}

const canvas = document.getElementById('neural-canvas');
const ctx = canvas.getContext('2d');
let W, H, ncx, ncy, t=0, nodes=[], pulses=[], pulseTimer=0;

function heartbeat(time) {
  const period = (60/55)*60;
  const phase = (time%period)/period;
  const lub = Math.exp(-Math.pow((phase-0.1)/0.04,2));
  const dub = Math.exp(-Math.pow((phase-0.2)/0.03,2))*0.6;
  return 1+(lub+dub)*0.22;
}
function resize() {
  W=canvas.width=window.innerWidth;
  H=canvas.height=window.innerHeight;
  ncx=W/2; ncy=H/2;
  buildNodes();
}
function buildNodes() {
  nodes=[];
  const maxR=Math.min(W,H)*0.33;
  const layers=[
    {n:1,r:0,speed:0,sz:3.1},{n:10,r:maxR*0.18,speed:0.005,sz:2.4},
    {n:18,r:maxR*0.35,speed:-0.004,sz:2.0},{n:24,r:maxR*0.52,speed:0.003,sz:1.6},
    {n:32,r:maxR*0.68,speed:-0.002,sz:1.3},{n:40,r:maxR*0.82,speed:0.0015,sz:1.1},
    {n:48,r:maxR*0.95,speed:-0.001,sz:0.9},{n:54,r:maxR*1.08,speed:0.0008,sz:0.7},
  ];
  layers.forEach(l => {
    for(let i=0;i<l.n;i++) {
      const a=(i/l.n)*Math.PI*2+Math.random()*0.3;
      nodes.push({baseAngle:a,r:l.r+(Math.random()-0.5)*l.r*0.15,speed:l.speed*(0.8+Math.random()*0.4),sz:l.sz+Math.random()*0.6,pulseOff:Math.random()*Math.PI*2,bright:0.4+Math.random()*0.6});
    }
  });
}
function pos(n,time) {
  const a=n.baseAngle+n.speed*time;
  const hb=heartbeat(time);
  return {x:ncx+Math.cos(a)*n.r*hb, y:ncy+Math.sin(a)*n.r*hb*0.9};
}
function dist(a,b){return Math.sqrt((a.x-b.x)**2+(a.y-b.y)**2);}
function draw() {
  ctx.clearRect(0,0,W,H);
  t++; pulseTimer++;
  const hb=heartbeat(t);
  const maxR=Math.min(W,H)*0.30;
  const positions=nodes.map(n=>pos(n,t));
  const atm=ctx.createRadialGradient(ncx,ncy,0,ncx,ncy,maxR*1.3);
  atm.addColorStop(0,`rgba(220,235,255,${0.025*hb*3})`);
  atm.addColorStop(0.5,`rgba(200,220,255,${0.025*hb})`);
  atm.addColorStop(1,'transparent');
  ctx.beginPath();ctx.arc(ncx,ncy,maxR*1.3,0,Math.PI*2);ctx.fillStyle=atm;ctx.fill();
  [0.18,0.35,0.52,0.68,0.82,0.95].forEach((rf,i)=>{
    const r=maxR*rf*hb, wave=Math.sin(t*0.008+i*1.1)*0.5+0.5;
    ctx.beginPath();ctx.arc(ncx,ncy,r,0,Math.PI*2);
    ctx.strokeStyle=`rgba(200,220,255,${0.04*wave})`;ctx.lineWidth=0.5;ctx.stroke();
  });
  const coreR=24*hb;
  const cg=ctx.createRadialGradient(ncx,ncy,0,ncx,ncy,coreR*3.5);
  cg.addColorStop(0,`rgba(255,255,255,${Math.min(1, 1.1*(hb-0.8)*5)})`);
  cg.addColorStop(0.3,`rgba(230,242,255,${0.55*hb})`);
  cg.addColorStop(0.6,`rgba(200,220,255,${0.25*hb})`);
  cg.addColorStop(1,'transparent');
  ctx.beginPath();ctx.arc(ncx,ncy,coreR*3.5,0,Math.PI*2);ctx.fillStyle=cg;ctx.fill();
  ctx.beginPath();ctx.arc(ncx,ncy,5.5*hb,0,Math.PI*2);
  ctx.fillStyle=`rgba(255,255,255,${Math.min(1,hb)})`;ctx.fill();
  const maxDist=maxR*0.55;
  for(let i=0;i<nodes.length;i++){
    for(let j=i+1;j<nodes.length;j++){
      const d=dist(positions[i],positions[j]);
      if(d<maxDist){
        ctx.beginPath();ctx.moveTo(positions[i].x,positions[i].y);ctx.lineTo(positions[j].x,positions[j].y);
        ctx.strokeStyle=`rgba(210,228,255,${(1-d/maxDist)*0.1*nodes[i].bright*hb})`;ctx.lineWidth=0.4;ctx.stroke();
      }
    }
    const d2=dist(positions[i],{x:ncx,y:ncy});
    if(d2<maxR*0.9){
      ctx.beginPath();ctx.moveTo(ncx,ncy);ctx.lineTo(positions[i].x,positions[i].y);
      ctx.strokeStyle=`rgba(230,240,255,${(1-d2/(maxR*0.9))*0.06*nodes[i].bright*hb})`;ctx.lineWidth=0.3;ctx.stroke();
    }
  }
  nodes.forEach((n,i)=>{
    const p=positions[i];
    const pSz=n.sz*(0.88+0.12*Math.sin(t*0.03+n.pulseOff))*hb;
    const br=n.bright*Math.min(hb,1.15);
    const halo=ctx.createRadialGradient(p.x,p.y,0,p.x,p.y,pSz*4);
    halo.addColorStop(0,`rgba(210,228,255,${0.12*br})`);halo.addColorStop(1,'transparent');
    ctx.beginPath();ctx.arc(p.x,p.y,pSz*4,0,Math.PI*2);ctx.fillStyle=halo;ctx.fill();
    ctx.beginPath();ctx.arc(p.x,p.y,pSz,0,Math.PI*2);
    ctx.fillStyle=`rgba(255,255,255,${br*0.88})`;ctx.fill();
  });
  if(pulseTimer%40===0){
    const a=nodes[Math.floor(Math.random()*nodes.length)];
    const b=nodes[Math.floor(Math.random()*nodes.length)];
    if(a!==b) pulses.push({a,b,p:0,sp:0.015+Math.random()*0.01});
  }
  pulses=pulses.filter(p=>p.p<1);
  pulses.forEach(p=>{
    p.p+=p.sp;
    const pa=pos(p.a,t),pb=pos(p.b,t);
    const x=pa.x+(pb.x-pa.x)*p.p,y=pa.y+(pb.y-pa.y)*p.p;
    const g=ctx.createRadialGradient(x,y,0,x,y,5);
    g.addColorStop(0,`rgba(255,255,255,${0.75*(1-p.p)*hb})`);g.addColorStop(1,'transparent');
    ctx.beginPath();ctx.arc(x,y,5,0,Math.PI*2);ctx.fillStyle=g;ctx.fill();
  });
  requestAnimationFrame(draw);
}

let currentScheduled = null;
let toastTimeout = null;
function checkSchedule() {
  const now = new Date();
  const h = now.getHours(), m = now.getMinutes();
  for(const item of scheduledItems) {
    if(item.hour === h && item.minute === m) {
      if(item.autoOpen) { openContent(item.panel); }
      else { showScheduledToast(item); }
      break;
    }
  }
}
function showScheduledToast(item) {
  currentScheduled = item;
  document.getElementById('toast-icon').textContent = item.icon;
  document.getElementById('toast-title').textContent = item.title;
  document.getElementById('toast-sub').textContent = item.sub;
  const toast = document.getElementById('scheduled-toast');
  toast.classList.add('show');
  if(toastTimeout) clearTimeout(toastTimeout);
  toastTimeout = setTimeout(dismissToast, 10000);
}
function dismissToast() {
  document.getElementById('scheduled-toast').classList.remove('show');
  if(toastTimeout) clearTimeout(toastTimeout);
}
function openScheduled() {
  dismissToast();
  if(currentScheduled) openContent(currentScheduled.panel);
}
